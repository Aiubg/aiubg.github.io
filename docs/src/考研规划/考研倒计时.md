    <script setup>
        // 1. 定义考研目标日期
        // 请将 '2025-12-20T00:00:00' 替换为你实际的考研日期和时间。
        // 格式为 'YYYY-MM-DDTHH:mm:ss'，T后面是时间，不写时间默认是当天零点。
        const targetDate = new Date('2025-12-20T00:00:00').getTime();

        // 2. 获取需要更新的 DOM 元素
        const daysEl = document.getElementById('days');
        const hoursEl = document.getElementById('hours');
        const minutesEl = document.getElementById('minutes');
        const secondsEl = document.getElementById('seconds');
        const titleEl = document.querySelector('.title'); // 获取标题元素
        const mottoEl = document.querySelector('.motto'); // 获取座右铭元素


        // 3. 创建一个函数来计算并更新倒计时
        function updateCountdown() {
            // 获取当前时间
            const now = new Date().getTime();

            // 计算剩余时间（毫秒）
            const timeLeft = targetDate - now;

            // 定义时间单位的毫秒值
            const second = 1000;
            const minute = second * 60;
            const hour = minute * 60;
            const day = hour * 24;

            // 检查是否已过目标日期
            if (timeLeft <= 0) {
                // 倒计时结束
                daysEl.innerText = '00';
                hoursEl.innerText = '00';
                minutesEl.innerText = '00';
                secondsEl.innerText = '00';
                if (titleEl) titleEl.innerText = '考研已结束！';
                 if (mottoEl) mottoEl.innerText = '祝你金榜题名！'; // 或其他结束语
                clearInterval(countdownInterval); // 停止计时器
                return; // 结束函数执行
            }

            // 计算剩余的天、小时、分钟、秒
            const days = Math.floor(timeLeft / day);
            const hours = Math.floor((timeLeft % day) / hour); // 剩余天数后的小时数
            const minutes = Math.floor((timeLeft % hour) / minute); // 剩余小时数后的分钟数
            const seconds = Math.floor((timeLeft % minute) / second); // 剩余分钟数后的秒数

            // 更新页面上的数字，并确保总是两位数显示 (例如：5 显示为 05)
            daysEl.innerText = String(days).padStart(2, '0');
            hoursEl.innerText = String(hours).padStart(2, '0');
            minutesEl.innerText = String(minutes).padStart(2, '0');
            secondsEl.innerText = String(seconds).padStart(2, '0');
        }

        // 4. 每秒更新一次倒计时
        // 使用 setInterval 函数，每隔 1000 毫秒 (1秒) 调用一次 updateCountdown 函数
        const countdownInterval = setInterval(updateCountdown, 1000);

        // 5. 页面加载后立即执行一次更新，避免等待第一秒
        updateCountdown();
    </script>
